tabela2 <- xdata %>%
  distinct(ID, .keep_all = TRUE) %>% 
  replace_with_na(replace = list(PCT = -999)) %>%
  select(HEIGHT, BW, AGE, SOFA, PCT, ELWI, CI, CRE, ALB, PROTEIN) %>%
  rename("Waga [kg]" = BW, "Prokalcytonina [ng/ml]" = PCT, "Wskaźnik pozanaczyniowej wody wewnątrzpłucnej" = ELWI, "Wiek [lat]" = AGE, "Wzrost [cm]" = HEIGHT, "Wskaźnik sercowy [L/min/m2]" = CI, "Kreatynina [mg/dL]" = CRE, "Albumina [g/dL]" = ALB, "Białko całkowite [mg/ml]" = PROTEIN) %>% 
  
  pivot_longer(cols = everything(), names_to = "Zmienna", values_to = "Wartość") %>%
  mutate(Zmienna = factor(Zmienna, levels = unique(Zmienna))) %>%
  group_by(Zmienna) %>% 
  summarise(
    `N (ilość)` = sum(!is.na(Wartość)),
    `Średnia` = round(mean(Wartość, na.rm = TRUE), 2),
    `Mediana` = round(median(Wartość, na.rm = TRUE), 2),
    `Min` = round(min(Wartość, na.rm = TRUE), 2),
    `Max` = round(max(Wartość, na.rm = TRUE), 2),
    `Odch. stand. (SD)` = round(sd(Wartość, na.rm = TRUE), 2)
  )

Kowarianty ciągłe wg zastosowania ECMO

ciag_ecmo <- xdata %>%
  filter(EVID == 1) %>%
  distinct(ID, .keep_all = TRUE) %>%
  mutate(ECMO = factor(ECMO, levels = c(0, 1), labels = c("Bez ECMO", "ECMO")))

cont_vars <- c("AGE", "BW", "SOFA", "PCT", "ELWI", "CI", "CRE", "ALB", "PROTEIN")

ciag_ECMO <- ciag_ecmo %>%
  select(ID, ECMO, all_of(cont_vars)) %>%
  pivot_longer(cols = all_of(cont_vars),
               names_to = "Zmienna",
               values_to = "Wartość") %>%
  mutate(Zmienna = factor(Zmienna, levels = cont_vars))

units_labels <- c(
  AGE = "AGE [lata]",
  BW  = "BW [kg]",
  SOFA  = "SOFA",
  PCT = "PCT [ng/mL]",
  ELWI = "ELWI",
  CI  = "CI [L/min/m²]",
  CRE = "CRE [mg/dL]",
  ALB = "ALB [g/dL]",
  PROTEIN = "PROTEIN [mg/dL]"
)

ciag_ECMO <- ciag_ECMO %>%
  mutate(Zmienna_label = factor(units_labels[Zmienna], levels = units_labels))

plot_ecmo <- ggplot(ciag_ECMO, aes(ECMO, Wartość)) +
  geom_boxplot(outlier.shape = NA, alpha = 0.7, width = 0.5) +
  facet_wrap(~ Zmienna_label, scales = "free_y", ncol = 3) +
  labs(
    title = "Kowarianty ciągłe wg zastosowania \n pozausrojowego natleniania krwi",
    y = "Wartość",
    x = NULL
  ) +
  theme_bw(base_size = 14)+
  theme(legend.position = "none")

print(plot_ecmo)
